FUNCOES E AGRUPAMENTOS

FUNCAO

SUN = SOMA
AVG = MEDIA
COUNT = QUANTIDADE
MAX = MAIOR VALOR
MIN = MENOR VALOR

====

FUNCAO SOMA

SELECT  COLUNA, "FUNCAO"(coluna)
    FROM TABELA
        WHERE CONDICAO
            ORDER BY COLUNA


EXEMPLO
    SELECT SUM(SALARIO)
        FROM VENDEDOR
            WHERE SALARIO >1000;

=======

FUNCAO QUANTIDADE

SELECT  COLUNA, "FUNCAO"(coluna)
    FROM TABELA
        WHERE CONDICAO
            ORDER BY COLUNA


EXEMPLO
    SELECT COUNT(*)
        FROM VENDEDOR
            WHERE SALARIO < 1000;            

=======


AGRUPAMENTOS = GROUP BY

SELECT  COLUNA, "FUNCAO"(coluna)
    FROM TABELA
        WHERE CONDICAO
            GROUP BY COLUNA


EXEMPLO
    SELECT NOME SUM(SALARIO)
        FROM VENDEDOR
            GROUP BY NOME


=======


AGRUPAMENTOS = HAVING (APLICA DO FILTRO APOS FAZER O CALCULO, DIFERENTE DO WHERE QUE FAZ O FILTRO ANTES)

SELECT  COLUNA, "FUNCAO"(coluna)
    FROM TABELA
        WHERE CONDICAO
            GROUP BY COLUNA
                HAVING CONIDICAO


EXEMPLO
    SELECT NOME SUM(SALARIO)
        FROM VENDEDOR
            GROUP BY NOME
            HAVING SUM(QNT)> 200
