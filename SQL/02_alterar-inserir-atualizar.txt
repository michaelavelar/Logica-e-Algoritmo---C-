DROP - APAGAR A ESTRUTURA TA TABELA

 DROP TABLE CLIENTE


 ===============

 RENAME - RENOMEAR TABELA 

 RENAME CLIENTE TO CLIENTE_2

 =============

 TRUNCATLE TABLE - REMOVE TODAS AS LINHAS DA TABELA
 
 TRUNCATE TABLE CLIENTE


===============

DELETE - APAGA DADOS DE UMA TABELA

DELETE FROM CLIENTE
    [WHERE NOME = '']


===============

UPDATE = ATUALIZAR DADOS DE UMA TABELA

UPDATE TABELA
SET COLUNA = NOVO_VALOR
[WHERE CONDICAO]

EXEMPLO:

UPDATE CLIENTE 
 SET NOME = 'JOSE'
 WHERE CPF = '01234567890'

 =========

 INSERT INTO - INSERIR DADOS EM UMA TABELA

 INSERT INSO TABELA(COLUNA1 [, COLUNA2...])
 VALUES (VALOR1 [, VALOR2...])

 OU

 INSERT INT TABELA 1
 SELECT CAMPOS FROM TABELA2

 EXEMPLO

 INSERT INTO CLIENTE(NOME, IDADE, CPF)
    VALUES('MICHAEL', 32, 111111111)
